197  lock_type lock( _mutex ) ;
198  auto typeIter = _pluginFactories.find( type ) ;
199  auto factoryIter = typeIter->second.find( name ) ;
200  // plugin not found ?
201  if ( typeIter->second.end() == factoryIter ) {
202  auto typeStr = pluginTypeToString( type ) ;

204  return nullptr ;
205  }
