368  inline const std::string ParameterT<T>::type() const {
369  // make this human readable
370  if ( typeid( _value ) == typeid( EVENT::IntVec )) return "IntVec" ;
371  else if( typeid( _value ) == typeid( EVENT::FloatVec )) return "FloatVec" ;
372  else if( typeid( _value ) == typeid( EVENT::StringVec )) return "StringVec" ;
373  else if( typeid( _value ) == typeid( int )) return "int" ;
374  else if( typeid( _value ) == typeid( float )) return "float" ;
375  else if( typeid( _value ) == typeid( double )) return "double" ;
376  else if( typeid( _value ) == typeid(std::string) ) return "string" ;
377  else if( typeid( _value ) == typeid( bool ) ) return "bool";
378  else return typeid( _value ).name() ;
379  }
